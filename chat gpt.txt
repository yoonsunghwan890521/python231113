# chat gpt를 활용한 파이썬 코드 (주석달기)

# 함수 정의: 인자가 없는 경우 1을 반환, 하나의 인자가 주어진 경우 해당 인자를 반환, 두 개의 인자가 주어진 경우 곱을 반환
def times(a=1, b=1):
    return a * b

# 함수 호출 및 출력
print(times())       # 아무 인자도 주어지지 않으면 기본값으로 1 * 1을 출력
print(times(5))      # 하나의 인자가 주어지면 해당 값과 기본값인 1을 곱하여 출력
print(times(5, 6))    # 두 개의 인자가 주어지면 두 값을 곱하여 출력

# 키워드 인자를 사용하여 서버와 포트를 조합하는 함수 정의
def connectURI(server, port):
    strURL = "http://" + server + ":" + port
    return strURL

# 함수 호출 및 출력
print(connectURI("multi.com", "80"))          # 위치 인자를 사용하여 호출
print(connectURI(port="8080", server="multi.com"))  # 키워드 인자를 사용하여 호출

# 가변 인자를 사용하여 여러 문자열을 하나로 합치는 함수 정의
def union(*ar):
    result = []
    for item in ar:
        for x in item:
            if x not in result:
                result.append(x)
    return result

# 함수 호출 및 출력
print(union("HAM", "EGG"))          # 두 문자열을 합쳐 중복을 제거한 결과 출력
print(union("HAM", "EGG", "SPAM"))  # 세 문자열을 합쳐 중복을 제거한 결과 출력
